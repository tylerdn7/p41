INCLUDE 'EMU8086.INC'
.MODEL SMALL
.STACK 100H
.DATA 

    NUMBER DB 1,2,3,4
    NUMBER2 DB 5 DUP()

.CODE

MAIN PROC
    
    MOV AX, @DATA
    MOV DS, AX
    
    LEA SI, NUMBER
    
    MOV CX, 5
    MOV AH, 2
    
    FOR:
        MOV DL, [SI]
        ADD DL, 48
        INT 21H
        ADD SI, 1
    LOOP FOR
    
    
    MOV AH, 4CH
    INT 21H
    MAIN ENDP 

    
END MAIN
